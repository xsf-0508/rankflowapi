Easy Integration

Copy, paste, and configure. It's that simple.

Step 1

Get Your Key

Purchase a license and use /setup in our Discord to generate your Secret Token.

Step 2

Configure Studio

Enable HTTP Requests in Game Settings > Security to allow external API calls.

Step 3

Deploy Script

Paste the script into a ServerScript in ServerScriptService.

ServerScriptService/RankHandler.lua Lua 
-- RankFlow API Integration Example
local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")

-- CONFIGURATION
-- Ensure this matches your Railway Endpoint
local API_URL = "https://rankflow-production-e079.up.railway.app/api/v2/rank/update" 
local SECRET_TOKEN = "YOUR_TOKEN_HERE" -- Get this via /setup command
local GROUP_ID = 35119105 
local MIN_RANK = 40

-- API Calling Function
local function callRankApi(targetId, action)
    local data = {
        userId = tostring(targetId),
        action = action
    }

    local success, response = pcall(function()
        return HttpService:RequestAsync({
            Url = API_URL,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json",
                ["Authorization"] = SECRET_TOKEN
            },
            Body = HttpService:JSONEncode(data)
        })
    end)

    if success and response.StatusCode == 200 then
        print("‚úÖ Ranking Successful!")
        return true
    else
        warn("‚ùå Request Failed: " .. (response and response.StatusCode or "Error"))
    end
    return false
end

-- Chat Command Example: !promote [username]
Players.PlayerAdded:Connect(function(player)
    player.Chatted:Connect(function(msg)
        if player:GetRankInGroup(GROUP_ID) >= MIN_RANK then
            local cmd, target = msg:match("^!(%w+)%s+(%S+)$")
            if cmd == "promote" or cmd == "demote" then
                local success, targetId = pcall(function()
                    return Players:GetUserIdFromNameAsync(target)
                end)
                
                if success then
                    callRankApi(targetId, cmd)
                end
            end
        end
    end)
end)
